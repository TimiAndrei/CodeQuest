def is_palindrome(x: int) -> bool:
    if x < 0:
        return False

    original = x
    reversed_num = 0

    while x != 0:
        digit = x % 10
        reversed_num = reversed_num * 10 + digit
        x //= 10

    return original == reversed_num

def main():
    import sys
    input = sys.stdin.read().strip()
    x = int(input)
    print(is_palindrome(x))

if __name__ == "__main__":
    main()
