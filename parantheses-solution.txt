function longestValidParentheses(s) {
    let maxLength = 0;
    let stack = [-1]; // Initialize stack with -1 to handle edge cases

    for (let i = 0; i < s.length; i++) {
        if (s[i] === '(') {
            stack.push(i); // Push the index of '(' onto the stack
        } else {
            stack.pop(); // Pop the last '(' index
            if (stack.length === 0) {
                stack.push(i); // Push the current index as a new base
            } else {
                maxLength = Math.max(maxLength, i - stack[stack.length - 1]);
            }
        }
    }

    return maxLength;
}

function main() {
    const fs = require('fs');
    const input = fs.readFileSync('/dev/stdin', 'utf8').trim();
    console.log(longestValidParentheses(input));
}

if (require.main === module) {
    main();
}
